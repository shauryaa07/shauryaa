âš™ STEP 3: JavaScript (script.js)

This connects the webcam and controls the mic.

const video = document.getElementById('localVideo');
const micBtn = document.getElementById('micBtn');
let stream;
let micOn = true;

// Step 1: Start camera & mic
async function startCamera() {
  try {
    stream = await navigator.mediaDevices.getUserMedia({
      video: true,
      audio: true
    });
    video.srcObject = stream;
  } catch (err) {
    console.error("Error accessing camera/mic:", err);
  }
}

// Step 2: Toggle mic on/off
micBtn.addEventListener('click', () => {
  if (!stream) return;
  const audioTrack = stream.getAudioTracks()[0];
  micOn = !micOn;
  audioTrack.enabled = micOn;
  micBtn.textContent = micOn ? 'ğŸ™' : 'ğŸ”‡';
});

// Start everything
startCamera();

  isDragging = true;
  offsetX = e.clientX - mini.getBoundingClientRect().left;
  offsetY = e.clientY - mini.getBoundingClientRect().top;
});

document.addEventListener('mouseup', () => isDragging = false);

document.addEventListener('mousemove', (e) => {
  if (!isDragging) return;
  mini.style.left = ${e.clientX - offsetX}px;
  mini.style.top = ${e.clientY - offsetY}px;
});
